command sword:
	trigger:
		if inventory of player contains sword:
			stop
		else:
			give player unbreakable wooden sword named "&7T1 Sword" with nbt "{HideFlags:5}"

on right click:
	if player is sneaking:
		if name of player's tool does not contain "&7T":
			stop
		if player's tool is sword:
			create a gui with virtual chest inventory with 4 rows named "&6&lSword Upgrades":
				make gui all integers between 0 and 36 with black glass pane named "&f"
				make gui 16 with sunflower named "&eTokens: &6%format({token::%player's uuid%})%"
				make gui 10 with glowing book named "&6+1 Sharpness" with lore "&7Cost: 500 Tokens" and "&7Max: 5":
					if {token::%player's uuid%} >= 500:
						remove 500 from {token::%player's uuid%}
						if level of Sharpness of player's tool >= 1:
							if level of Sharpness of player's tool <= 4:
								set {_lore::*} to lore of player's tool
								enchant player's tool with "Sharpness %(level of Sharpness of player's tool + 1)%" parsed as enchantment type
								remove "&7- Sharpness %(level of Sharpness of player's tool - 1)%" from {_lore::*}
								add "&7- Sharpness %level of Sharpness of player's tool%" to {_lore::*}
								set lore of player's tool to {_lore::*}
							else:
								send "&cSharpness is maxed!"
						else:
							set {_lore::*} to lore of player's tool
							enchant player's tool with Sharpness 1
							if {_lore::*} doesn't contain "&6&lENCHANTMENTS":
								add "%nl%&6&lENCHANTMENTS" to {_lore::*}
							add "&7- Sharpness 1" to {_lore::*}
							set lore of player's tool to {_lore::*}
				
			
				make gui 11 with glowing book named "&6+1 Knockback" with lore "&7Cost: 750 Tokens" and "&7Max: 3":
					if {token::%player's uuid%} >= 750:
						remove 750 from {token::%player's uuid%}
						if level of Knockback of player's tool >= 1:
							if level of Knockback of player's tool <= 2:
								set {_lore::*} to lore of player's tool
								enchant player's tool with "Knockback %(level of Knockback of player's tool + 1)%" parsed as enchantment type
								remove "&7- Knockback %(level of Knockback of player's tool - 1)%" from {_lore::*}
								add "&7- Knockback %level of Knockback of player's tool%" to {_lore::*}
								set lore of player's tool to {_lore::*}
							else:
								send "&cKnockback is maxed!"
						else:
							set {_lore::*} to lore of player's tool
							enchant player's tool with knockback 1
							if {_lore::*} doesn't contain "&6&lENCHANTMENTS":
								add "%nl%&6&lENCHANTMENTS" to {_lore::*}
							add "&7- Knockback 1" to {_lore::*}
							set lore of player's tool to {_lore::*}
					
						
				make gui 12 with glowing book named "&6+1 Sweeping Edge" with lore "&7Cost: 1000 Tokens" and "&7Max: 5":
					if {token::%player's uuid%} >= 1000:
						remove 1000 from {token::%player's uuid%}
						if level of sweeping edge of player's tool >= 1:
							if level of sweeping edge of player's tool <= 4:
								set {_lore::*} to lore of player's tool
								enchant player's tool with "Sweeping Edge %(level of Sweeping Edge of player's tool + 1)%" parsed as enchantment type
								remove "&7- Sweeping Edge %(level of Sweeping Edge of player's tool - 1)%" from {_lore::*}
								add "&7- Sweeping Edge %level of Sweeping Edge of player's tool%" to {_lore::*}
								set lore of player's tool to {_lore::*}
							else:
								send "&cSweeping Edge is maxed!"
						else:
							set {_lore::*} to lore of player's tool
							enchant player's tool with Sweeping Edge 1
							if {_lore::*} doesn't contain "&6&lENCHANTMENTS":
								add "%nl%&6&lENCHANTMENTS" to {_lore::*}
							add "&7- Sweeping Edge 1" to {_lore::*}
							set lore of player's tool to {_lore::*}
				make gui 19 with glowing book named "&6+1 Fire Aspect" with lore "&7Cost: 2500 Tokens" and "&7Max: 3":
					if {token::%player's uuid%} >= 2500:
						remove 2500 from {token::%player's uuid%}
						if level of Fire Aspect of player's tool >= 1:
							if level of Fire Aspect of player's tool <= 2:
								set {_lore::*} to lore of player's tool
								enchant player's tool with "Fire Aspect %(level of Fire Aspect of player's tool + 1)%" parsed as enchantment type
								remove "&7- Fire Aspect %(level of Fire Aspect of player's tool - 1)%" from {_lore::*}
								add "&7- Fire Aspect %level of Fire Aspect of player's tool%" to {_lore::*}
								set lore of player's tool to {_lore::*}
							else:
								send "&cFire Aspect is maxed!"
						else:
							set {_lore::*} to lore of player's tool
							enchant player's tool with Fire Aspect 1
							if {_lore::*} doesn't contain "&6&lENCHANTMENTS":
								add "%nl%&6&lENCHANTMENTS" to {_lore::*}
							add "&7- Fire Aspect 1" to {_lore::*}
							set lore of player's tool to {_lore::*}		
						
						
						
						
						
						
						
						
						
						
						
						
						
						
				make gui 20 with glowing oak sign named "&c&lMore Enchantments coming soon"
				make gui 15 with paper named "&bPrestige your Sword!" with lore "&4Warning: This will reset your enchants." and "&cCost: 5000 Tokens & 3 Sharpness":
					if {token::%player's uuid%} >= 5000:
						if player's tool is netherite sword:
							send "&6&lPizzaGenz &eYou have maxed your prestige for your sword."
							stop
						remove 1000 from {token::%player's uuid%}
						send "&aYou have prestiged your sword!"
					else:
						send "&cYou cannot afford this!"
						stop
					set {_e} to level of sharpness of player's tool
					if {_e} >= 3:
						close player's inventory
						if player's tool is wooden Sword:
							remove player's tool from player's inventory
							give player unbreakable stone Sword named "&7T2 Sword" with nbt "{HideFlags:5}"
						else if player's tool is stone Sword:
							remove player's tool from player's inventory
							give player unbreakable iron Sword named "&7T3 Sword" with nbt "{HideFlags:5}"
						else if player's tool is iron Sword:
							remove player's tool from player's inventory
							give player unbreakable diamond Sword named "&7T4 Sword" with nbt "{HideFlags:5}"
						else if player's tool is diamond Sword:
							remove player's tool from player's inventory
							give player unbreakable netherite sword named "&7T5 Sword" with nbt "{HideFlags:5}"
							
						else if player's tool is any pickaxe:
							send "&6&lPizzaGenz &eYou cannot prestige this tool."
					else:
						send "&6&lPizzaGenz &eYou do not have the required enchants to prestige your pickaxe"
			open last gui to player
